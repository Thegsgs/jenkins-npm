+	numRetries := 0
+	for numRetries <= maxRetries {
+		numRetries++
+		connection, err := Connect(id)

+			if numRetries >= maxRetries {
+				//log.Fatal calls os.Exit(1), so the service will exit in this case

+					Msg("Failed to open activeMQ connection - max retries reached - exiting") //TODO pass message in func arg
+			}
+			time.Sleep(time.Duration(backoffSeconds) * time.Second)
+			log.Error().Err(err).Int("retry", numRetries).Str("Connection purpose", connectionPurpose).
+				Msg("Failed to open activeMQ connection") //TODO pass message in func arg
+		} else {
+			log.Info().Str("Connection purpose", connectionPurpose).Msg("Created activeMQ connection successfully")
